        def get_kvm_definition(vmID)
                # call login to get the link to VMs
                url = "http://#{@@username}:#{@@password}@#{@@server}/api/login"
                login = _httpget(url)
                login['link'].each do |x|
                        if x['rel'] == 'virtualmachines'
                                url = x['href']
                        end
                end
                vms = _httpget(url)
                $log.info "VMs count: #{vms.length}"
        end

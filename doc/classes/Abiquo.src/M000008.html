        def get_vdc_by_id(id)
                url = "http://#{@@username}:#{@@password}@#{@@server}/api/cloud/virtualdatacenters"
                xml = self._httpget(url)

                xml['virtualDatacenter'].each { |vdc|
                        if vdc["id"][0] == id
                                $log.debug "Found VDC #{vdc['id']}"
                                return vdc
                        end
                }

                return nil
        end

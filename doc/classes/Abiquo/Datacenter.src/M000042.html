        def get_racks()
                retracks = Array.new()
                racksxml = RestClient::Request.new(:method => :get, :url => @racks, :user => @@username, :password => @@password).execute
                r = Nokogiri::XML.parse(racksxml).xpath('//racks/rack')
                r.each do |rack|
                        retracks << Abiquo::Rack.new(rack)
                end

                return retracks
        end

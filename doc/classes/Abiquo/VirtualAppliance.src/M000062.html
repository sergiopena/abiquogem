        def delete()
                $log.debug "Trying to delte #{url}"
                self.get_state
                case @state[0]
                when "NOT_DEPLOYED"
                        RestClient.delete(@url)
                        $log.info "Deleted #{url}"
                when "DEPLOYED"
                        $log.info "Vapps is deployed #{url}"
                        self.undeploy
                        RestClient.delete(@url)

                end
        end
